164  OptionalJacobian<3, 3> H = boost::none) {
165  assert(xyz.size() == 3);
166  Rot3 out;
167  if (H) {
168  Vector3 Hx, Hy, Hz;
169  out = RzRyRx(xyz(0), xyz(1), xyz(2), Hx, Hy, Hz);
170  (*H) << Hx, Hy, Hz;
171  } else
172  out = RzRyRx(xyz(0), xyz(1), xyz(2));
